function doGet(e) { return handleRequest(e); }
function doPost(e) { return handleRequest(e); }

function handleRequest(e) {
  var output = ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.JSON);

  var ss = SpreadsheetApp.getActiveSpreadsheet();

  var txSheet = ss.getSheetByName('Transactions');
  if (!txSheet) {
    txSheet = ss.insertSheet('Transactions');
    txSheet.appendRow(['ID','User ID','Type','Amount','Category','Reason','Date','Time','UPI ID','Payment App']);
  }

  var userSheet = ss.getSheetByName('Users');
  if (!userSheet) {
    userSheet = ss.insertSheet('Users');
    userSheet.appendRow(['ID','Name','Initial Savings','Running Income','Running Expense','Running Savings']);
  }

  var catSheet = ss.getSheetByName('Categories');
  if (!catSheet) {
    catSheet = ss.insertSheet('Categories');
    catSheet.appendRow(['Type','Name','Emoji']);
  }

  try {
    var rawData = '';
    if (e && e.postData && e.postData.contents) rawData = e.postData.contents;
    else if (e && e.parameter && e.parameter.data) rawData = e.parameter.data;

    var data = rawData ? JSON.parse(rawData) : {};
    var txs = data.transactions || [];
    var users = data.users || [];
    var deletedIds = data.deletedIds || [];
    var customCats = data.customCats || null;

    if (deletedIds.length > 0) {
      var existing = txSheet.getDataRange().getValues();
      var delMap = {};
      for (var d = 0; d < deletedIds.length; d++) { delMap[String(deletedIds[d])] = true; }
      for (var r = existing.length - 1; r > 0; r--) {
        if (delMap[String(existing[r][0])]) { txSheet.deleteRow(r + 1); }
      }
    }

    if (users.length > 0) {
      if (userSheet.getLastRow() > 1) userSheet.getRange(2, 1, userSheet.getLastRow() - 1, 6).clearContent();
      for (var i = 0; i < users.length; i++) {
        userSheet.appendRow([
          users[i].id, users[i].name, users[i].initialSavings,
          users[i].runningIncome || 0, users[i].runningExpense || 0, users[i].runningSavings || 0
        ]);
      }
    }

    if (txs.length > 0) {
      var currentData = txSheet.getDataRange().getValues();
      var existingIds = {};
      for (var r = 1; r < currentData.length; r++) { existingIds[String(currentData[r][0])] = true; }
      for (var i = txs.length - 1; i >= 0; i--) {
        if (!existingIds[String(txs[i].id)]) {
          txSheet.appendRow([
            txs[i].id, txs[i].userId, txs[i].type, txs[i].amount,
            txs[i].category, txs[i].reason, txs[i].date, txs[i].time,
            txs[i].upiId || '', txs[i].paymentPlatform || ''
          ]);
          existingIds[String(txs[i].id)] = true;
        }
      }
    }

    if (customCats) {
      if (catSheet.getLastRow() > 1) catSheet.getRange(2, 1, catSheet.getLastRow() - 1, 3).clearContent();
      var inc = customCats.income || [];
      for (var i = 0; i < inc.length; i++) catSheet.appendRow(['income', inc[i].label, inc[i].emoji]);
      var exp = customCats.expense || [];
      for (var i = 0; i < exp.length; i++) catSheet.appendRow(['expense', exp[i].label, exp[i].emoji]);
    }

    var finalTx = txSheet.getDataRange().getValues();
    var allTxs = [];
    for (var r = 1; r < finalTx.length; r++) {
      allTxs.push({
        id: String(finalTx[r][0]), userId: String(finalTx[r][1]),
        type: finalTx[r][2], amount: Number(finalTx[r][3]),
        category: finalTx[r][4], reason: finalTx[r][5],
        date: finalTx[r][6], time: finalTx[r][7],
        upiId: finalTx[r][8] || '', paymentPlatform: finalTx[r][9] || ''
      });
    }

    var finalUsers = userSheet.getDataRange().getValues();
    var allUsers = [];
    for (var r = 1; r < finalUsers.length; r++) {
      allUsers.push({
        id: String(finalUsers[r][0]), name: finalUsers[r][1],
        initialSavings: Number(finalUsers[r][2]),
        runningIncome: Number(finalUsers[r][3]),
        runningExpense: Number(finalUsers[r][4]),
        runningSavings: Number(finalUsers[r][5])
      });
    }

    var finalCats = catSheet.getDataRange().getValues();
    var allCats = { income: [], expense: [] };
    for (var r = 1; r < finalCats.length; r++) {
      var tType = finalCats[r][0], tName = finalCats[r][1], tEmoji = finalCats[r][2];
      if (tType && tName) {
        allCats[tType === 'income' ? 'income' : 'expense'].push({ label: tName, emoji: tEmoji, key: tName });
      }
    }

    output.setContent(JSON.stringify({ status: 'success', transactions: allTxs, users: allUsers, customCats: allCats }));
    return output;

  } catch (err) {
    output.setContent(JSON.stringify({ status: 'error', message: err.toString() }));
    return output;
  }
}
