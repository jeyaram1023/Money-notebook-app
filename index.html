Configuration Guide

1 Create Spreadsheet

Open Google Sheets and create a Blank spreadsheet.

2 Add the Script

Click Extensions > Apps Script. Delete existing code, then paste:

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var txSheet = ss.getSheetByName('Transactions');
  if (!txSheet) {
    txSheet = ss.insertSheet('Transactions');
    txSheet.appendRow(['ID', 'User ID', 'Type', 'Amount', 'Category', 'Reason', 'Date', 'Time', 'UPI ID', 'Payment App']);
  }
  var userSheet = ss.getSheetByName('Users');
  if (!userSheet) {
    userSheet = ss.insertSheet('Users');
    userSheet.appendRow(['ID', 'Name', 'Initial Savings', 'Running Income', 'Running Expense', 'Running Savings']);
  }

  try {
    var data = e.postData && e.postData.contents ? JSON.parse(e.postData.contents) : {transactions: [], users: [], deletedIds: []};
    var txs = data.transactions || [];
    var users = data.users || [];
    var deletedIds = data.deletedIds || [];

    if (deletedIds.length > 0) {
      var existingData = txSheet.getDataRange().getValues();
      var delMap = {};
      for(var d = 0; d < deletedIds.length; d++) { delMap[String(deletedIds[d])] = true; }
      for (var r = existingData.length - 1; r > 0; r--) {
        if (delMap[String(existingData[r][0])]) { txSheet.deleteRow(r + 1); }
      }
    }
    
    if (users.length > 0) {
      if (userSheet.getLastRow() > 1) { userSheet.getRange(2, 1, userSheet.getLastRow() - 1, 6).clearContent(); }
      for (var i = 0; i < users.length; i++) {
        userSheet.appendRow([users[i].id, users[i].name, users[i].initialSavings, users[i].runningIncome || 0, users[i].runningExpense || 0, users[i].runningSavings || 0]);
      }
    }
    
    if (txs.length > 0) {
        var currentData = txSheet.getDataRange().getValues();
        var existingIds = {};
        for(var r = 1; r < currentData.length; r++) { existingIds[String(currentData[r][0])] = true; }
        for (var i = txs.length - 1; i >= 0; i--) {
          if(!existingIds[String(txs[i].id)]) {
            txSheet.appendRow([txs[i].id, txs[i].userId, txs[i].type, txs[i].amount, txs[i].category, txs[i].reason, txs[i].date, txs[i].time, txs[i].upiId || '', txs[i].paymentPlatform || '']);
            existingIds[String(txs[i].id)] = true;
          }
        }
    }

    var finalTx = txSheet.getDataRange().getValues();
    var allTxs = [];
    for(var r = 1; r < finalTx.length; r++) {
      allTxs.push({ id: String(finalTx[r][0]), userId: String(finalTx[r][1]), type: finalTx[r][2], amount: Number(finalTx[r][3]), category: finalTx[r][4], reason: finalTx[r][5], date: finalTx[r][6], time: finalTx[r][7], upiId: finalTx[r][8] || '', paymentPlatform: finalTx[r][9] || '' });
    }

    var finalUsers = userSheet.getDataRange().getValues();
    var allUsers = [];
    for(var r = 1; r < finalUsers.length; r++) {
      allUsers.push({ id: String(finalUsers[r][0]), name: finalUsers[r][1], initialSavings: Number(finalUsers[r][2]), runningIncome: Number(finalUsers[r][3]), runningExpense: Number(finalUsers[r][4]), runningSavings: Number(finalUsers[r][5]) });
    }
    
    return ContentService.createTextOutput(JSON.stringify({"status": "success", "transactions": allTxs, "users": allUsers})).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

3 Deploy & Connect

  • Click Deploy > New deployment. Select Web app.
  • Set 'Execute as' to Me, and 'Who has access' to Anyone.
  • Click Deploy, copy Web App URL, and paste in Settings!
